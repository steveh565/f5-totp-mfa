1. Solution Overview

This guide provides complete instructions to deploy a TOTP (Time-Based One-Time Password) multi-factor authentication solution on F5 BIG-IP. The solution is entirely self-contained — all components run on the BIG-IP with no external dependencies.

2.5 Rate Limiting

Verification: 5 failed attempts per user per 5 minutes → lockout Enrollment: 3 failed confirmations per user per 10 minutes → lockout IP Throttle: 6 enrollment page loads per IP per 10 minutes → lockout All counters stored in subtables (TMM memory). Successful verification/enrollment resets the user's counter. Admin can reset via API or Admin UI. Rate limit parameters configurable in totp_config_dg data group.

TOTP authentication is time-sensitive. A clock drift of more than 30 seconds will cause all TOTP codes to fail.

# Verify synchronization
ntpq -p
# Look for * indicating active sync source
If no server shows * after several minutes, check firewall rules for NTP (UDP 123) and verify DNS resolution.

TOTP_ENC_KEY=$(openssl rand -base64 32 | tr -d '\n' | cut -c1-32)

7.5 Create Enrollment Persistence Script

cat > /config/totp/totp_enroll.sh << 'SCRIPT_EOF'
#!/bin/bash
#
# totp_enroll.sh - Persist TOTP secret to data group on enrollment
#
# Triggered by: logger -p local0.alert "TOTP_ENROLL_TRIGGER:<username>"
# Reads username from log, queries API for secret, updates data group
#

LOGFILE="/var/log/ltm"
API_HOST="totp-api.mfa-demo.local"
API_PORT="443"
DG_NAME="totp_secrets_dg"

# Get API key from config data group
API_KEY=$(tmsh list ltm data-group internal totp_config_dg records 2>/dev/null | grep -A1 "api_key" | grep "data" | awk '{print $2}' | tr -d '"')

if [ -z "$API_KEY" ]; then
    logger -p local0.err "totp_enroll: Failed to retrieve API key from totp_config_dg"
    exit 1
fi

# Extract username from most recent trigger in log
USERNAME=$(grep "TOTP_ENROLL_TRIGGER:" "$LOGFILE" | tail -1 | sed 's/.*TOTP_ENROLL_TRIGGER://' | awk '{print $1}')

if [ -z "$USERNAME" ]; then
    logger -p local0.err "totp_enroll: No username found in trigger"
    exit 1
fi

# Query API for encrypted secret
RESPONSE=$(curl -sk -H "X-API-Key: $API_KEY" \
    "https://${API_HOST}:${API_PORT}/api/v1/export-secret?username=${USERNAME}" 2>/dev/null)

SECRET=$(echo "$RESPONSE" | grep -o '"secret":"[^"]*"' | cut -d'"' -f4)

if [ -z "$SECRET" ]; then
    logger -p local0.err "totp_enroll: Failed to retrieve secret for user $USERNAME"
    exit 1
fi

# Update data group
tmsh modify ltm data-group internal "$DG_NAME" records add { "$USERNAME" { data "$SECRET" } } 2>/dev/null

if [ $? -ne 0 ]; then
    # Record may already exist, try modify
    tmsh modify ltm data-group internal "$DG_NAME" records modify { "$USERNAME" { data "$SECRET" } } 2>/dev/null
    if [ $? -ne 0 ]; then
        logger -p local0.err "totp_enroll: Failed to commit secret for user $USERNAME"
        exit 1
    fi
fi

# Save configuration
tmsh save sys config partitions { Common } 2>/dev/null
logger -p local0.info "totp_enroll: Committed secret for user $USERNAME to $DG_NAME"

exit 0
SCRIPT_EOF

chmod 700 /config/totp/totp_enroll.sh

8. iRules — Complete Code Listings

Create iRules via GUI: Local Traffic → iRules → iRule List → Create

iRules must be created in order: irule_totp_shared first, as other iRules depend on its procedures.

8.1 irule_totp_shared

Shared procedures for TOTP operations. Attach FIRST on all TOTP virtual servers.

# =============================================================================
# iRule: irule_totp_shared
# Purpose: Shared procedures for TOTP operations
# Attach FIRST on: vs_totp_enroll, vs_totp_api, vs_totp_api_internal,
#                  vs_mfa_portal, vs_totp_admin
# =============================================================================

# ---- Configuration Data Group Name ----
proc totp_config_dg_name {} { return "totp_config_dg" }

# ---- Subtable and Data Group Names ----
proc totp_users_subtable {} { return "totp_users" }
proc totp_rate_subtable {} { return "totp_ratelimit" }
proc totp_enroll_rate_subtable {} { return "totp_enroll_ratelimit" }
proc totp_secrets_dg_name {} { return "totp_secrets_dg" }

# ---- Config Lookups ----
proc totp_get_api_key {} {
    set k [class match -value "api_key" equals [call totp_config_dg_name]]
    if { $k eq "" } { log local0.err "TOTP: api_key not found" }
    return $k
}

proc totp_get_encryption_key {} {
    set k [class match -value "encryption_key" equals [call totp_config_dg_name]]
    if { $k eq "" } { log local0.err "TOTP: encryption_key not found" }
    return $k
}

proc totp_get_issuer {} {
    set v [class match -value "issuer" equals [call totp_config_dg_name]]
    if { $v eq "" } { return "MFA Demo" }
    return $v
}

proc totp_get_period {} {
    set v [class match -value "time_step" equals [call totp_config_dg_name]]
    if { $v ne "" } { return $v }
    return 30
}

proc totp_get_window {} {
    set v [class match -value "time_skew" equals [call totp_config_dg_name]]
    if { $v ne "" } { return $v }
    return 1
}

proc totp_rate_get_max {} {
    set v [class match -value "rate_max_attempts" equals [call totp_config_dg_name]]
    if { $v ne "" } { return $v }
    return 5
}

proc totp_rate_get_window {} {
    set v [class match -value "rate_window_seconds" equals [call totp_config_dg_name]]
    if { $v ne "" } { return $v }
    return 300
}

proc totp_enroll_rate_get_max {} {
    set v [class match -value "enroll_rate_max_attempts" equals [call totp_config_dg_name]]
    if { $v ne "" } { return $v }
    return 3
}

proc totp_enroll_rate_get_window {} {
    set v [class match -value "enroll_rate_window_seconds" equals [call totp_config_dg_name]]
    if { $v ne "" } { return $v }
    return 600
}

# ---- Two-Tier Secret Lookup (Lazy Loading) ----
proc totp_lookup_secret { username } {
    set u [string tolower $username]
    # Try subtable first (fast path)
    set c [table lookup -subtable [call totp_users_subtable] $u]
    if { $c ne "" } { return $c }
    # Fall back to data group (slow path)
    set s [class match -value $u equals [call totp_secrets_dg_name]]
    if { $s ne "" } {
        # Cache in subtable for future requests
        table set -subtable [call totp_users_subtable] $u $s indefinite indefinite
        return $s
    }
    return ""
}

proc totp_is_enrolled { username } {
    if { [call totp_lookup_secret $username] ne "" } { return 1 }
    return 0
}

proc totp_store_secret { username encrypted_secret } {
    set u [string tolower $username]

# ---- Encryption / Decryption ----
proc totp_encrypt_secret { plaintext } {
    set k [call totp_get_encryption_key]
    if { $k eq "" } { return "" }
    if { [catch {
        set e [CRYPTO::encrypt -alg aes-256-ecb -key [b64decode $k] $plaintext]
    } err] } {
        log local0.err "TOTP: Encrypt failed: $err"
        return ""
    }
    return $e
}

proc totp_decrypt_secret { ciphertext } {
    set k [call totp_get_encryption_key]
    if { $k eq "" } { return "" }
    if { [catch {
        set d [CRYPTO::decrypt -alg aes-256-ecb -key [b64decode $k] $ciphertext]
    } err] } {
        log local0.err "TOTP: Decrypt failed: $err"
        return ""
    }
    return $d
}

# ---- Binary/Integer Conversion (Tcl 8.5 Compatible) ----
proc bits_to_int { bits } {
    set result 0
    foreach bit [split $bits ""] {
        set result [expr { ($result << 1) | $bit }]
    }
    return $result
}

proc int_to_bits { num width } {
    set bits ""
    for { set i [expr {$width - 1}] } { $i >= 0 } { incr i -1 } {
        append bits [expr { ($num >> $i) & 1 }]
    }
    return $bits
}

# ---- Base32 Encode / Decode ----
proc totp_base32_decode { b32_string } {
    set alphabet "ABCDEFGHIJKLMNOPQRSTUVWXYZ234567"
    set bits ""
    set clean [string trimright [string toupper $b32_string] "="]
    foreach char [split $clean ""] {
        set idx [string first $char $alphabet]
        if { $idx < 0 } { continue }
        append bits [call int_to_bits $idx 5]
    }
    set bytes ""
    set bitlen [string length $bits]
    for { set i 0 } { [expr { $i + 7 }] < $bitlen } { incr i 8 } {
        set byte_bits [string range $bits $i [expr { $i + 7 }]]
        append bytes [format %c [call bits_to_int $byte_bits]]
    }
    return $bytes
}

proc totp_bytes_to_base32 { bytes } {
    set alphabet "ABCDEFGHIJKLMNOPQRSTUVWXYZ234567"
    set b32 ""
    set bits ""
    foreach byte [split $bytes ""] {
        binary scan $byte c signed_val
        set unsigned_val [expr { $signed_val & 0xff }]
        append bits [call int_to_bits $unsigned_val 8]
    }
    set bitlen [string length $bits]
    set pad [expr { (5 - ($bitlen % 5)) % 5 }]
    append bits [string repeat "0" $pad]
    for { set i 0 } { $i < [string length $bits] } { incr i 5 } {
        set chunk [string range $bits $i [expr { $i + 4 }]]
        set idx [call bits_to_int $chunk]
        append b32 [string index $alphabet $idx]
    }
    return $b32
}

proc totp_generate_secret {} {
    set rb [AES::key 256]
    return [call totp_bytes_to_base32 [string range $rb 0 19]]
}

# ---- HOTP / TOTP ----
proc totp_compute_hotp { secret_bytes counter } {
    # Format counter as 8-byte big-endian
    set high [expr { ($counter >> 32) & 0xffffffff }]
    set low [expr { $counter & 0xffffffff }]
    set cb [binary format II $high $low]
    
    set hmac [CRYPTO::sign -alg hmac-sha1 -key $secret_bytes $cb]
    binary scan $hmac H* hx
    
    # Get offset from last nibble
    scan [string range $hx end end] %x last_nibble
    set off [expr { $last_nibble & 0x0f }]
    set s [expr { $off * 2 }]
    
    # Extract 4 bytes and compute code
    scan [string range $hx $s [expr { $s + 7 }]] %x dbc
    return [format %06d [expr { ($dbc & 0x7fffffff) % 1000000 }]]
}

proc totp_verify_raw { secret_b32 code } {
    set sb [call totp_base32_decode $secret_b32]
    set now [clock seconds]
    set period [call totp_get_period]
    set window [call totp_get_window]
    for { set i [expr { -1 * $window }] } { $i <= $window } { incr i } {
        set ctr [expr { ($now + ($i * $period)) / $period }]
        if { $code eq [call totp_compute_hotp $sb $ctr] } { return 1 }
    }
    return 0
}

# ---- Verification Rate Limiting ----
proc totp_rate_is_blocked { username } {
    set u [string tolower $username]
    set key "rate_${u}"
    set max [call totp_rate_get_max]
    set subtable [call totp_rate_subtable]
    set c [table lookup -subtable $subtable $key]
    if { $c ne "" && $c >= $max } {
        set r [table timeout -subtable $subtable -remaining $key]
        log local0.warn "TOTP_RATE:BLOCKED:${u}:attempts=${c},max=${max},remaining=${r}s"
        return 1
    }
    return 0
}

proc totp_rate_increment { username } {
    set u [string tolower $username]
    set key "rate_${u}"
    set max [call totp_rate_get_max]
    set window [call totp_rate_get_window]
    set subtable [call totp_rate_subtable]
    set c [table lookup -subtable $subtable $key]
    if { $c eq "" } {
        table set -subtable $subtable $key 1 $window $window
        log local0.info "TOTP_RATE:ATTEMPT:${u}:attempt=1,max=${max}"
        return 1
    }
    set nc [expr { $c + 1 }]
    set r [table timeout -subtable $subtable -remaining $key]
    if { $r < 1 } { set r $window }
    table set -subtable $subtable $key $nc $r $r
    if { $nc >= $max } {
        log local0.alert "TOTP_RATE:LOCKOUT:${u}:attempts=${nc},max=${max},locked_for=${r}s"
    } else {
        log local0.info "TOTP_RATE:ATTEMPT:${u}:attempt=${nc},max=${max}"
    }
    return $nc
}

# ---- Rate-Limited TOTP Verification ----
proc totp_verify_code_rated { username code } {
    set u [string tolower $username]
    if { [call totp_rate_is_blocked $username] } { return "rate_limited" }
    set an [call totp_rate_increment $username]
    set enc [call totp_lookup_secret $username]
    if { $enc eq "" } { return "not_enrolled" }
    set sec [call totp_decrypt_secret $enc]
    if { $sec eq "" } { return "decrypt_error" }
    if { [call totp_verify_raw $sec $code] } {
        call totp_rate_reset $username
        return "success"
    }
    return "invalid_code"
}

# ---- Bulk Rate Reset ----
proc totp_rate_bulk_reset { admin_info } {
    set vc 0
    set ec 0
    set rate_st [call totp_rate_subtable]
    set enroll_st [call totp_enroll_rate_subtable]

        table delete -subtable $rate_st $key

        table delete -subtable $enroll_st $key
        incr ec
    }

    log local0.notice "TOTP_RATE:BULK_RESET:verify=${vc},enroll=${ec},admin=${admin_info}"
    return [list $vc $ec $t]
}

        } else {

                "Wrong code. Try again." "/reenroll" "&#x2190; Try Again"] \
                "Content-Type" "text/html; charset=utf-8"
        }
        return
    }

    # Re-enrollment: confirm new code (POST /reenroll/confirm)
    if { [HTTP::path] eq "/reenroll/confirm" } {
        if { $username eq "" } {
            HTTP::respond 403 content "Access denied" "Content-Type" "text/plain"
            return
        }
        set ns [ACCESS::session data get "session.custom.totp.reenroll_new_secret"]
        set vf [ACCESS::session data get "session.custom.totp.reenroll_verified"]
        if { $ns eq "" || $vf ne "yes" } {

                "Please start re-enrollment again." "/reenroll" "&#x2190; Start Over"] \
                "Content-Type" "text/html; charset=utf-8"
            return
        }
        set tc ""
        foreach pair [split $payload "&"] {
            set kv [split $pair "="]
            if { [lindex $kv 0] eq "totp_code" } {
                set tc [string trim [lindex $kv 1]]
            }
        }
        set tc [string map {" " "" "-" ""} $tc]
        if { ![regexp {^[0-9]{6}$} $tc] } {
            HTTP::respond 200 content [call enroll_status "Invalid Code" "&#x274C;" "#da3633" \
                "Enter exactly 6 digits." "/reenroll" "&#x2190; Try Again"] \
                "Content-Type" "text/html; charset=utf-8"
            return
        }
        if { [call irule_totp_shared::totp_verify_raw $ns $tc] } {
            set encrypted [call irule_totp_shared::totp_encrypt_secret $ns]
            if { $encrypted eq "" } {
                HTTP::respond 200 content [call enroll_status "Error" "&#x274C;" "#da3633" \
                    "Internal error." "/reenroll" "&#x2190; Try Again"] \
                    "Content-Type" "text/html; charset=utf-8"
                return
            }
            call irule_totp_shared::totp_store_secret $username $encrypted
            ACCESS::session data set "session.custom.totp.reenroll_new_secret" ""
            ACCESS::session data set "session.custom.totp.reenroll_verified" ""

            HTTP::respond 200 content [call enroll_status "Re-enrollment Complete!" \
                "&#x2705;" "#1a7f37" "Your TOTP authenticator has been replaced." "" ""] \
                "Content-Type" "text/html; charset=utf-8" "Cache-Control" "no-store"
        } else {
            HTTP::respond 200 content [call enroll_status "Invalid Code" "&#x274C;" "#da3633" \
                "Wrong code. Try again." "/reenroll" "&#x2190; Try Again"] \
                "Content-Type" "text/html; charset=utf-8"
        }
        return
    }

    # Original enrollment confirm
    set secret [ACCESS::session data get "session.custom.totp.enroll_secret"]
    if { $username eq "" || $secret eq "" } {
        HTTP::respond 200 content [call enroll_status "Session Expired" "&#x274C;" "#da3633" \
            "Please start enrollment again." "/enroll" "&#x2190; Start Over"] \
            "Content-Type" "text/html; charset=utf-8"

            "Content-Type" "text/html; charset=utf-8"

    }
    set tc ""
    foreach pair [split $payload "&"] {
        set kv [split $pair "="]
        if { [lindex $kv 0] eq "totp_code" } {
            set tc [string trim [lindex $kv 1]]
        }
    }

        set encrypted [call irule_totp_shared::totp_encrypt_secret $secret]
        if { $encrypted eq "" } {
            HTTP::respond 200 content [call enroll_status "Error" "&#x274C;" "#da3633" \
                "Internal error." "/enroll" "&#x2190; Try Again"] \
                "Content-Type" "text/html; charset=utf-8"
            return
        }
        call irule_totp_shared::totp_store_secret $username $encrypted
        ACCESS::session data set "session.custom.totp.enroll_secret" ""
        call irule_totp_shared::totp_enroll_rate_reset $username
        log local0.info "TOTP-ENROLL: Complete for $username"
        HTTP::redirect "/enroll/complete"
    } else {
        set mx [call irule_totp_shared::totp_enroll_rate_get_max]
        set ra [expr { $mx - $an }]
        if { $ra < 0 } { set ra 0 }
        log local0.warn "TOTP-ENROLL: Invalid code $username ($an/$mx)"
        if { $ra > 0 } {
            HTTP::respond 200 content [call enroll_status "Invalid Code" "&#x274C;" "#da3633" \
                "Wrong code. $ra attempt(s) remaining." "/enroll" "&#x2190; Return to Enrollment"] \
                "Content-Type" "text/html; charset=utf-8"
        } else {
            set lt [call irule_totp_shared::totp_enroll_rate_remaining $username]
            HTTP::respond 200 content [call enroll_status "Enrollment Locked" "&#x1F512;" "#da3633" \
                "Try again in $lt seconds." "" ""] \
                "Content-Type" "text/html; charset=utf-8"
        }
    }
}

8.3 irule_totp_api

REST API for TOTP verification and management.

# =============================================================================
# iRule: irule_totp_api
# Virtual Servers: vs_totp_api (10.1.1.101:443), vs_totp_api_internal (10.255.255.255:80)
# Requires: irule_totp_shared attached first
# =============================================================================

proc api_json { status body } {
    HTTP::respond $status content $body \
        "Content-Type" "application/json" \
        "Cache-Control" "no-store" \
        "X-Content-Type-Options" "nosniff"
}

proc api_json_retry { status body ra } {
    HTTP::respond $status content $body \
        "Content-Type" "application/json" \
        "Cache-Control" "no-store" \
        "X-Content-Type-Options" "nosniff" \
        "Retry-After" $ra
}

proc api_dispatch_result { result username } {
    set ul [string tolower $username]
    switch $result {
        "success" {
            call api_json 200 "\{\"success\":true,\"username\":\"$ul\"\}"
        }
        "not_enrolled" {
            call api_json 401 "\{\"success\":false,\"username\":\"$ul\",\"error\":\"not_enrolled\"\}"
        }
        "decrypt_error" {
            call api_json 500 "\{\"success\":false,\"error\":\"internal_error\"\}"
        }
        "invalid_code" {
            set mx [call irule_totp_shared::totp_rate_get_max]
            set ra [expr { $mx - [call irule_totp_shared::totp_rate_get_count $username] }]
            if { $ra < 0 } { set ra 0 }
            call api_json 401 "\{\"success\":false,\"username\":\"$ul\",\"error\":\"invalid_code\",\"remaining_attempts\":$ra\}"
        }
        "rate_limited" {
            set r [call irule_totp_shared::totp_rate_remaining $username]
            call api_json_retry 429 "\{\"success\":false,\"username\":\"$ul\",\"error\":\"rate_limited\",\"retry_after\":$r\}" $r
        }
    }
}

when HTTP_REQUEST {
    set path [HTTP::path]
    set meth [HTTP::method]
    
    # API key validation
    set ek [call irule_totp_shared::totp_get_api_key]
    set pk [HTTP::header value "X-API-Key"]
    
    if { $ek eq "" } {
        call api_json 500 "\{\"error\":\"server_misconfiguration\"\}"
        event disable all
        return
    }
    if { $pk ne $ek } {
        call api_json 401 "\{\"error\":\"unauthorized\",\"message\":\"Invalid or missing API key\"\}"
        event disable all
        return
    }

    # GET /api/v1/health
    if { $path eq "/api/v1/health" } {
        call api_json 200 "\{\"status\":\"ok\"\}"
        event disable all
        return
    }

    # GET /api/v1/check?username=...
    if { $path eq "/api/v1/check" } {
        set u [URI::query [HTTP::uri] "username"]
        if { $u eq "" } {
            call api_json 400 "\{\"error\":\"missing_username\"\}"
            event disable all
            return
        }
        set ul [string tolower $u]
        if { [call irule_totp_shared::totp_is_enrolled $u] } {
            call api_json 200 "\{\"enrolled\":true,\"username\":\"$ul\"\}"
        } else {
            call api_json 200 "\{\"enrolled\":false,\"username\":\"$ul\"\}"
        }
        event disable all
        return
    }

    # GET /api/v1/status?username=...
    if { $path eq "/api/v1/status" } {
        set u [URI::query [HTTP::uri] "username"]
        if { $u eq "" } {
            call api_json 400 "\{\"error\":\"missing_username\"\}"
            event disable all
            return
        }
        set ul [string tolower $u]
        set c [call irule_totp_shared::totp_rate_get_count $u]
        set b [call irule_totp_shared::totp_rate_is_blocked $u]
        set r [call irule_totp_shared::totp_rate_remaining $u]
        set mx [call irule_totp_shared::totp_rate_get_max]
        set remaining [expr { $mx - $c }]
        if { $remaining < 0 } { set remaining 0 }
        call api_json 200 "\{\"username\":\"$ul\",\"attempts\":$c,\"max_attempts\":$mx,\"remaining\":$remaining,\"blocked\":$b,\"retry_after\":$r\}"
        event disable all
        return
    }

    # GET /api/v1/verify?username=...&code=...
    if { $path eq "/api/v1/verify" && $meth eq "GET" } {
        set u [URI::query [HTTP::uri] "username"]
        set c [URI::query [HTTP::uri] "code"]
        if { $u eq "" || $c eq "" } {
            call api_json 400 "\{\"error\":\"missing_parameters\"\}"
            event disable all
            return
        }
        set c [string map {" " "" "-" ""} [string trim $c]]
        if { ![regexp {^[0-9]{6}$} $c] } {
            call api_json 400 "\{\"error\":\"invalid_code_format\"\}"
            event disable all
            return
        }
        call api_dispatch_result [call irule_totp_shared::totp_verify_code_rated $u $c] $u
        event disable all
        return
    }

    # POST /api/v1/verify
    if { $path eq "/api/v1/verify" && $meth eq "POST" } {
        set cl 0
        catch { set cl [HTTP::header value "Content-Length"] }
        if { $cl > 0 && $cl < 4096 } {
            HTTP::collect $cl
        } else {
            call api_json 400 "\{\"error\":\"invalid_content_length\"\}"
            event disable all
        }
        return
    }

    # POST /api/v1/rate-reset?username=...
    if { $path eq "/api/v1/rate-reset" && $meth eq "POST" } {
        set u [URI::query [HTTP::uri] "username"]
        if { $u eq "" } {
            call api_json 400 "\{\"error\":\"missing_username\"\}"
            event disable all
            return
        }
        call irule_totp_shared::totp_rate_reset $u
        call api_json 200 "\{\"success\":true,\"username\":\"[string tolower $u]\",\"message\":\"Rate limit reset\"\}"
        event disable all
        return
    }

    # POST /api/v1/rate-reset-all
    if { $path eq "/api/v1/rate-reset-all" && $meth eq "POST" } {
        set c [call irule_totp_shared::totp_rate_bulk_reset "api_client=[IP::client_addr]"]
        call api_json 200 "\{\"success\":true,\"message\":\"All rate limits cleared\",\"cleared\":\{\"verify\":[lindex $c 0],\"enroll\":[lindex $c 1],\"total\":[lindex $c 2]\}\}"
        event disable all
        return
    }

    # GET /api/v1/export-secret?username=...
    if { $path eq "/api/v1/export-secret" } {
        set u [URI::query [HTTP::uri] "username"]
        if { $u eq "" } {
            call api_json 400 "\{\"error\":\"missing_username\"\}"
            event disable all
            return
        }
        set ul [string tolower $u]
        set s [call irule_totp_shared::totp_lookup_secret $u]
        if { $s eq "" } {
            call api_json 404 "\{\"error\":\"not_found\",\"username\":\"$ul\"\}"
        } else {
            call api_json 200 "\{\"username\":\"$ul\",\"secret\":\"$s\"\}"
        }
        event disable all
        return
    }

    # POST /api/v1/populate-table
    if { $path eq "/api/v1/populate-table" && $meth eq "POST" } {
        set dg_name [call irule_totp_shared::totp_secrets_dg_name]
        set subtable [call irule_totp_shared::totp_users_subtable]
        set count 0
        foreach record [class names $dg_name] {
            set secret [class match -value $record equals $dg_name]
            if { $secret ne "" } {
                table set -subtable $subtable $record $secret indefinite indefinite
                incr count
            }
        }
        log local0.info "TOTP: Populated subtable with $count users from $dg_name"
        call api_json 200 "\{\"success\":true,\"message\":\"Table populated\",\"users_loaded\":$count\}"
        event disable all
        return
    }

    call api_json 404 "\{\"error\":\"not_found\"\}"
    event disable all
}

when HTTP_REQUEST_DATA {
    set payload [HTTP::payload]
    set ct [HTTP::header value "Content-Type"]
    set u ""
    set c ""
    
    if { [string match -nocase "*json*" $ct] } {
        regexp {"username"\s*:\s*"([^"]*)"} $payload -> u
        regexp {"code"\s*:\s*"([^"]*)"} $payload -> c
    } else {
        foreach pair [split $payload "&"] {
            set kv [split $pair "="]
            switch -- [URI::decode [lindex $kv 0]] {
                "username" { set u [URI::decode [lindex $kv 1]] }
                "code" { set c [URI::decode [lindex $kv 1]] }
            }
        }
    }
    
    if { $u eq "" || $c eq "" } {
        call api_json 400 "\{\"error\":\"missing_parameters\"\}"
        return
    }
    set c [string map {" " "" "-" ""} [string trim $c]]
    if { ![regexp {^[0-9]{6}$} $c] } {
        call api_json 400 "\{\"error\":\"invalid_code_format\"\}"
        return
    }
    if { ![regexp {^[a-zA-Z0-9._@-]+$} $u] } {
        call api_json 400 "\{\"error\":\"invalid_username\"\}"
        return
    }
    call api_dispatch_result [call irule_totp_shared::totp_verify_code_rated $u $c] $u
}

        call irule_totp_shared::totp_rate_bulk_reset "admin_ui"
        log local0.info "ADMIN: Bulk rate reset"
        HTTP::redirect "/"
        return
    }

    # Unenroll user
    if { $path eq "/unenroll" && $username ne "" } {
        call irule_totp_shared::totp_delete_secret $username
        log local0.info "ADMIN: Unenrolled $username"
        HTTP::redirect "/users"
        return
    }

    HTTP::redirect "/"
}
